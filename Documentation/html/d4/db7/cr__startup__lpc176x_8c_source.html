00214 //*****************************************************************************
00215 // Functions to carry out the initialization of RW and BSS data sections. These
00216 // are written as separate functions rather than being inlined within the
00217 // ResetISR() function in order to cope with MCUs with multiple banks of
00218 // memory.
00219 //*****************************************************************************
00220 __attribute__ ((section(".after_vectors")))
00221 void data_init(unsigned int romstart, unsigned int start, unsigned int len) {
00222         unsigned int *pulDest = (unsigned int*) start;
00223         unsigned int *pulSrc = (unsigned int*) romstart;
00224         unsigned int loop;
00225         for (loop = 0; loop < len; loop = loop + 4)
00226                 *pulDest++ = *pulSrc++;
00227 }
00228 
00229 __attribute__ ((section(".after_vectors")))
00230 void bss_init(unsigned int start, unsigned int len) {
00231         unsigned int *pulDest = (unsigned int*) start;
00232         unsigned int loop;
00233         for (loop = 0; loop < len; loop = loop + 4)
00234                 *pulDest++ = 0;
00235 }

00269 //*****************************************************************************
00270 // Reset entry point for your code.
00271 // Sets up a simple runtime environment and initializes the C/C++
00272 // library.
00273 //*****************************************************************************
00274 __attribute__ ((section(".after_vectors")))
00275 void
00276 ResetISR(void) {
00277 
00278 #ifndef USE_OLD_STYLE_DATA_BSS_INIT
00279     //
00280     // Copy the data sections from flash to SRAM.
00281     //
00282         unsigned int LoadAddr, ExeAddr, SectionLen;
00283         unsigned int *SectionTableAddr;
00284 
00285         // Load base address of Global Section Table
00286         SectionTableAddr = &__data_section_table;
00287 
00288     // Copy the data sections from flash to SRAM.
00289         while (SectionTableAddr < &__data_section_table_end) {
00290                 LoadAddr = *SectionTableAddr++;
00291                 ExeAddr = *SectionTableAddr++;
00292                 SectionLen = *SectionTableAddr++;
00293                 data_init(LoadAddr, ExeAddr, SectionLen);
00294         }
00295         // At this point, SectionTableAddr = &__bss_section_table;
00296         // Zero fill the bss segment
00297         while (SectionTableAddr < &__bss_section_table_end) {
00298                 ExeAddr = *SectionTableAddr++;
00299                 SectionLen = *SectionTableAddr++;
00300                 bss_init(ExeAddr, SectionLen);
00301         }
00302 #else
00303         // Use Old Style Data and BSS section initialization.
00304         // This will only initialize a single RAM bank.
00305         unsigned int * LoadAddr, *ExeAddr, *EndAddr, SectionLen;
00306 
00307     // Copy the data segment from flash to SRAM.
00308         LoadAddr = &_etext;
00309         ExeAddr = &_data;
00310         EndAddr = &_edata;
00311         SectionLen = (void*)EndAddr - (void*)ExeAddr;
00312         data_init((unsigned int)LoadAddr, (unsigned int)ExeAddr, SectionLen);
00313         // Zero fill the bss segment
00314         ExeAddr = &_bss;
00315         EndAddr = &_ebss;
00316         SectionLen = (void*)EndAddr - (void*)ExeAddr;
00317         bss_init ((unsigned int)ExeAddr, SectionLen);
00318 #endif
00319 
00320 #ifdef __USE_CMSIS
00321         SystemInit();
00322 #endif
00323 
00324 #if defined (__cplusplus)
00325         //
00326         // Call C++ library initialisation
00327         //
00328         __libc_init_array();
00329 #endif
00330 
00331 #if defined (__REDLIB__)
00332         // Call the Redlib library, which in turn calls main()
00333         __main() ;
00334 #else
00335         main();
00336 #endif
00337 
00338         //
00339         // main() shouldn't return, but if it does, we'll just enter an infinite loop
00340         //
00341         while (1) {
00342                 ;
00343         }
00344 }
