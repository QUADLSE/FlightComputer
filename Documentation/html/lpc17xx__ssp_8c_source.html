00057 static void setSSPclock (LPC_SSP_TypeDef *SSPx, uint32_t target_clock)
00058 {
00059     uint32_t prescale, cr0_div, cmp_clk, ssp_clk;
00060 
00061     CHECK_PARAM(PARAM_SSPx(SSPx));
00062 
00063     /* The SSP clock is derived from the (main system oscillator / 2),
00064        so compute the best divider from that clock */
00065     if (SSPx == LPC_SSP0){
00066         ssp_clk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_SSP0);
00067     } else if (SSPx == LPC_SSP1) {
00068         ssp_clk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_SSP1);
00069     } else {
00070         return;
00071     }
00072 
00073         /* Find closest divider to get at or under the target frequency.
00074            Use smallest prescale possible and rely on the divider to get
00075            the closest target frequency */
00076         cr0_div = 0;
00077         cmp_clk = 0xFFFFFFFF;
00078         prescale = 2;
00079         while (cmp_clk > target_clock)
00080         {
00081                 cmp_clk = ssp_clk / ((cr0_div + 1) * prescale);
00082                 if (cmp_clk > target_clock)
00083                 {
00084                         cr0_div++;
00085                         if (cr0_div > 0xFF)
00086                         {
00087                                 cr0_div = 0;
00088                                 prescale += 2;
00089                         }
00090                 }
00091         }
00092 
00093     /* Write computed prescaler and divider back to register */
00094     SSPx->CR0 &= (~SSP_CR0_SCR(0xFF)) & SSP_CR0_BITMASK;
00095     SSPx->CR0 |= (SSP_CR0_SCR(cr0_div)) & SSP_CR0_BITMASK;
00096     SSPx->CPSR = prescale & SSP_CPSR_BITMASK;
00097 }
